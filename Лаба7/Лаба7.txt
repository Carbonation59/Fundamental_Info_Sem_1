//Перетаскивание крайней цифры числа к крайней цифре маски//
0*0->00*
0*1->10*
1*0->01*
1*1->11*
//Запись результата//
00*|->|
10*|->|0
01*|->|
11*|->|1

//Стирание числа//
0$|->$|
1$|->$|


//Начало результата//
00*->|
10*->|0
01*->|
11*->|1
//Маркер крайней цифры числа//
0$->$0*
1$->$1*
//Cтирание маски//
0|->|
1|->|
//Убираем незначащие нули//
$|0->$|
$|1->1
$|->0
